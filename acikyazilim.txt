using System;

class Program
{
    static void Main()
    {
        Console.WriteLine("Sayı Bulma Oyununa Hoş Geldiniz!");
        Console.WriteLine("Lütfen 1 ile 100 arasında bir sayı girin.");

        Random random = new Random();
        int bilgisayarSayisi = random.Next(1, 101);

        int tahmin;
        int denemeSayisi = 0;
        bool sayiBulundu = false;

        do
        {
            Console.Write("Tahmininizi girin: ");

        
            if (int.TryParse(Console.ReadLine(), out tahmin))
            {
                denemeSayisi++;

                if (tahmin == bilgisayarSayisi)
                {
                    sayiBulundu = true;
                }
                else if (tahmin < bilgisayarSayisi)
                {
                    Console.WriteLine("Daha büyük bir sayı girin.");
                }
                else
                {
                    Console.WriteLine("Daha küçük bir sayı girin.");
                }
            }
            else
            {
                Console.WriteLine("Geçerli bir sayı girişi yapmadınız. Lütfen tekrar deneyin.");
            }

        } while (!sayiBulundu);

        Console.WriteLine($"Tebrikler! {denemeSayisi} denemede sayıyı buldunuz. Sayı: {bilgisayarSayisi}");
    }
}
